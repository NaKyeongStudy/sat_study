    - 올림 함수를 따로 만들어서 함수 안에서 함수부르기

    -   ```python
        def main():
            mid = input("Enter grade on midterm: ")
            final = input("Enter grade on final exam: ")
            print("Semester Grade: ", end="")
            semesterGrade(mid, final)

        def semesterGrade(mid, final):
            average = (mid + (final*2)) / 3
            ave = ceil(average)
            if ave >= 90:
                print('A')
            elif ave >= 80:
                print('B')
            elif ave >= 70:
                print('C')
            elif ave >= 60:
                print('D')
            else:
                print('F')

        def ceil(num):
            if int(num) != num:
                return int(num) + 1
            else:
                return num

        main()
    '''
